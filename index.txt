        function handleKeyPress(event) {
            if (gameBlocked) return;
            
            let newX = playerPosition.x;
            let newY = playerPosition.y;
            
            switch(event.key) {
                case 'ArrowUp':
                    newY = Math.max(0, playerPosition.y - 1);
                    event.preventDefault();
                    break;
                case 'ArrowDown':
                    newY = Math.min(15, playerPosition.y + 1);
                    event.preventDefault();
                    break;
                case 'ArrowLeft':
                    newX = Math.max(0, playerPosition.x - 1);
                    event.preventDefault();
                    break;
                case 'ArrowRight':
                    newX = Math.min(19, playerPosition.x + 1);
                    event.preventDefault();
                    break;
                default:
                    return;
            }
            
            movePlayer(newX, newY);
        }
        
        function movePlayerByButton(direction) {
            if (gameBlocked) return;
            
            let newX = playerPosition.x;
            let newY = playerPosition.y;
            
            switch(direction) {
                case 'up':
                    newY = Math.max(0, playerPosition.y - 1);
                    break;
                case 'down':
                    newY = Math.min(15, playerPosition.y + 1);
                    break;
                case 'left':
                    newX = Math.max(0, playerPosition.x - 1);
                    break;
                case 'right':
                    newX = Math.min(19, playerPosition.x + 1);
                    break;
                default:
                    return;
            }
            
            movePlayer(newX, newY);
        }
        
        function movePlayer(newX, newY) {
            const currentMaze = mazeLayouts[currentDifficulty];
            const cellType = currentMaze[newY][newX];
            
            // Check if move is valid (adjacent cell)
            const deltaX = Math.abs(newX - playerPosition.x);
            const deltaY = Math.abs(newY - playerPosition.y);
            if (deltaX + deltaY !== 1) return; // Only allow one step moves

if (moveSound) {
    moveSound.currentTime = 0;
    moveSound.volume = 0.6;
    moveSound.play().catch(() => {});
}
            
            // Remove player from current position
            const currentCell = document.querySelector(`[data-x="${playerPosition.x}"][data-y="${playerPosition.y}"]`);
            currentCell.classList.remove('player');
            const currentCellType = currentMaze[playerPosition.y][playerPosition.x];
            
           if (currentCellType === 2) {
    currentCell.className = 'maze-cell start';
    currentCell.innerHTML = '<div style="position:absolute; top:2px; left:2px; font-size:9px; font-weight:bold;">START</div>';
} else if (currentCellType === 1) {
    currentCell.className = 'maze-cell path';
    currentCell.innerHTML = '';
} else if (currentCellType === 14) {
    currentCell.className = 'maze-cell obstacle';
    currentCell.innerHTML = `<img src="assets/images/black bomb.png" alt="Bomb">`;
} else if (currentCellType >= 3 && currentCellType <= 12) {
    const data = checkpointData[currentCellType];
    currentCell.className = 'maze-cell checkpoint';
    if (completedCheckpoints.has(currentCellType)) {
        currentCell.classList.add('completed-checkpoint');
    }
    currentCell.innerHTML = `
        <div style="position:absolute; top:2px; left:2px; font-size:9px; font-weight:bold; z-index:3;">${data.num}</div>
        <img src="assets/images/star 7.png" alt="Checkpoint" style="width:78%; height:78%; object-fit:contain;">
    `;
}
            
            
            // Move to new position
            playerPosition = { x: newX, y: newY };
            const newCell = document.querySelector(`[data-x="${newX}"][data-y="${newY}"]`);
            
           if (cellType === 0 || cellType === 14) {

    let image, message;

        image = newCell.dataset.dangerSrc || "assets/images/black bomb.png";

        image = newCell.dataset.dangerSrc || "assets/images/color croc 5.png";

    if (splashSound) {
        splashSound.currentTime = 0;
        splashSound.volume = 0.75;
        splashSound.play().catch(() => {});
    }

    setTimeout(() => {
        if (uhohSound) {
            uhohSound.currentTime = 0;
            uhohSound.volume = 0.8;
            uhohSound.play().catch(() => {});
        }
    }, 120);

    setTimeout(() => {
        if (gameOverSound) {
            gameOverSound.currentTime = 0;
            gameOverSound.volume = 0.8;
            gameOverSound.play().catch(() => {});
        }
    }, 260);

    showDangerModal(image, message);

    setTimeout(() => {
        newCell.classList.remove('engulfed');
        resetPlayerPosition();
    }, 1000);
}else if (cellType >= 3 && cellType <= 12) {
                // Hit checkpoint
                newCell.classList.add('player');
                
                if (completedCheckpoints.has(cellType)) {
                    // Show Moses on completed checkpoint but keep trophy visible
                    newCell.innerHTML = '<div style="font-size: 0.6em; line-height: 1;">' + checkpointData[cellType].num + '</div><div style="font-size: 0.8em;"><img src="assets/images/moses in basket.png" style="width:100%;height:100%;"></div>';
                } else {
                    // Show Moses on uncompleted checkpoint
                   newCell.innerHTML = `<img src="assets/images/moses in basket.png" alt="Moses" style="width:94%; height:94%; object-fit:contain;">`;
                    gameBlocked = true; // Block movement until puzzle is solved

if (checkpointSound) {
    checkpointSound.currentTime = 0;
    checkpointSound.volume = 0.75;
    checkpointSound.play().catch(() => {});
}


                    showPuzzleModal(cellType);
                }
                
            } else if (cellType === 13) {
                // Reached finish
                newCell.classList.add('player');
                newCell.innerHTML = '<img src="assets/images/moses in basket.png" alt="Moses" style="width:100%;height:100%;object-fit:contain;">';
                
                if (completedCheckpoints.size === 10) {
                    setTimeout(() => showWinModal(), 500);
                } else {
                    alert(`You need to visit all 10 checkpoints first! You've completed ${completedCheckpoints.size}/10.`);
                }
                
            } else {
                // Safe path
                newCell.classList.add('player');
                newCell.innerHTML = '<img src="assets/images/moses in basket.png" alt="Moses" style="width:100%;height:100%;object-fit:contain;">';

function checkAnswer() {
    const userAnswer = document.getElementById('answerInput').value.trim().toUpperCase();
    const correctAnswer = puzzles[currentPuzzle].answer;

    if (isAnswerCorrect(userAnswer, correctAnswer)) {
        if (correctSound) {
            correctSound.currentTime = 0;
            correctSound.volume = 0.8;
            correctSound.play().catch(() => {});
        }

        showFlashMessage("That's Correct! Great Job!", 'success');

            updateGameInfo();

            const checkpointCell = document.querySelector(`[data-x="${playerPosition.x}"][data-y="${playerPosition.y}"]`);
            checkpointCell.classList.add('completed-checkpoint');
            checkpointCell.innerHTML =

    } else {

            document.getElementById('answerInput').focus();
            document.getElementById('puzzleStatus').textContent =
                "You didn't get it quite right. Please try again or use a clue for help.";
        }, 1500);
    }
}




        
        function isAnswerCorrect(userAnswer, correctAnswer) {
            // Direct match
            if (userAnswer === correctAnswer) return true;
            
            // Handle close variations for each answer
            switch(correctAnswer) {
                case 'CRYING':
                    return userAnswer === 'CRY' || userAnswer === 'CRYING';
                case 'REEDS':
                    return userAnswer === 'REED' || userAnswer === 'REEDS';
                case 'GOD':
                    return userAnswer === 'GOD' || userAnswer === 'JESUS' || userAnswer === 'HOLY SPIRIT';
                case 'HEBREW':
                    return userAnswer === 'HEBREW' || userAnswer === 'ISRAELITE' || userAnswer === 'ISRAELITES';
                default:
                    return false;
            }
        }
